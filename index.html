Local Search - Duration
					
						fn improving_move(
							input: &DurationMatrix,
							tour: &Tour,
							tour_cost: u64,
						) -> Option<Move> {
							let mut best_cost = tour_cost;
							let mut best_move = None;

							for mv in feasible_moves(input, tour) {
								if mv.cost() < best_cost {
									(best_cost, best_move) = (mv.cost(), Some(mv));
								}
							}

							best_move
						}

						fn local_search(input: &DurationMatrix, mut tour: Tour) -> Tour {
							let mut cost = input.tour_cost(&tour);

							while let Some(mv) = improving_move(input, &tour, cost) {
								mv.apply(&mut tour);
								cost = mv.cost();
							}
							
							tour
